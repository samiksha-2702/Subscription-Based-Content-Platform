Class & Object

Class is a blueprint, object is an instance.

class Student {
public:
    string name;
    void show(){
        cout << name;
    }
};

Student s;
s.name = "Amit";
s.show();

Inheritance

class A{
public:
    void showA(){}
};

class B: public A{
public:
    void showB(){}
};
Flexibility

Polymorphism

class Base{
public:
    virtual void show(){
        cout<<"Base";
    }
};

class Derived: public Base{
public:
    void show() override{
        cout<<"Derived";
    }
};
Security

Encapsulation

class Bank{
private:
    int balance;

public:
    void setBalance(int b){
        balance=b;
    }

    int getBalance(){
        return balance;
    }
};
