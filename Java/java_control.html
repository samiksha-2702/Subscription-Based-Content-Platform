Java Control Flow

If Else

int age = 18;
if(age >= 18){
 System.out.println("Adult");
}else{
 System.out.println("Minor");
}

Switch

int day = 2;
switch(day){
 case 1: System.out.println("Mon"); break;
 case 2: System.out.println("Tue"); break;
 default: System.out.println("Invalid");
}

For Loop

for(int i=1;i<=5;i++){
 System.out.println(i);
}

While Loop

int i=1;
while(i<=5){
 System.out.println(i);
 i++;
}

Break & Continue

for(int i=1;i<=10;i++){
 if(i==5) break;
 if(i==2) continue;
}
